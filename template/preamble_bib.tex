\newcommand{\onlinecite}[2][]{\footnote{online: \citeauthor{#2}, \citeyear[][#1]{#2}\swallowarialspace{}\citeurldate{#2}}}

\DeclareNameAlias{default}{last-first}

\DeclareBibliographyDriver{online}{
    \printnames{author}
    \addcolon\space\newblock
    \printfield{title}
    \newunit\newblock
    \printlist{publisher}
    \newunit
    \printlist{location}
    \addcomma\space
    \printfield{year}
    \addcomma\space\newblock
    \printfield{url}
    \printurldate
}

\DeclareBibliographyDriver{book}{
    \printnames{author}
    \addcolon\space\newblock
    \printfield{title}
    \newunit\newblock
    \printlist{publisher}
    \newunit
    \printlist{location}
    \addcomma\space
    \printfield{year}
}

\DeclareBibliographyDriver{report}{
    \printnames{author}
    \addcolon\space\newblock
    \printfield{title}
    \addcolon\space
    \printlist{institution}
    \newunit
    \printlist{location}
    \addcomma\space
    \printfield{year}
    \addcomma\space
    \printfield{url}
}
