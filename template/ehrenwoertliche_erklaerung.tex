,,\ifthenelse{\value{authorcount} = 1}{Ich erkläre}{Wir erklären}\space

\begin{enumerate}
\item[1.]  
dass \ifthenelse{\value{authorcount} = 1}{ich meine}{wir unsere}\space
\reporttype \space mit dem Thema:

,,\reporttopic``

ohne fremde Hilfe angefertigt habe\ifthenelse{\value{authorcount} > 1}{n},

\item[2.] dass\space
\ifthenelse{\value{authorcount} = 1}{ich}{wir}\space

\item[3.] dass\space
\ifthenelse{\value{authorcount} = 1}{ich}{wir}\space
diese \reporttype\space bei keiner anderen Prüfung vorgelegt habe\ifthenelse{\value{authorcount} > 1}{n}.

\end{enumerate}
